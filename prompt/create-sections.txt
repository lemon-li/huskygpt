Here is a simple example of how to write a "sections":
import { useModel } from 'umi';

const Welcome: React.FC = () => {
  const { user, fetchUser } = useModel({{model file name under the "models"}}, model => ({ user: model.user, fetchUser: model.fetchUser }));

  return <div>user</div>
};

export default Welcome;

Please note that "sections" is a module of "pages"
