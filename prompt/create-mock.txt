
Here is a simple example of how to write a "mock":
import { Request, Response } from 'express';

// mock logic functions
async function getFakeCaptcha(req: Request, res: Response) {
  // some custom logic
  await waitTime(2000);
  return res.json('captcha-xxx');
}

// mock api responses
export default {
  // Complexity case by user requirements
  'GET /api/currentUser': (req: Request, res: Response) => {
    if (!getAccess()) {
      // res.status(401).send({{response json}});
      return;
    }
    // res.send({{response json}});
  },
  // Simple case by user requirements
  'GET /api/users': [
    {
      key: '1',
      name: 'John Brown',
      age: 32,
      address: 'New York No. 1 Lake Park',
    },
    {...}
  ],
};

Please note that you need send response json by following structure, the "data" field type T I will provide later:
interface ResponseStructure<T = any> {
  success: boolean;
  data: T;
  errorCode?: number;
  errorMessage?: string;
  showType?: ErrorShowType;
}
